# Создайте файл .env в этой папке со следующим содержимым:

BOT_TOKEN=ваш_токен_бота_здесь
OPENAI_API_KEY=ваш_openai_ключ_здесь  
ADMIN_CHAT_ID=ваш_админ_id_здесь

# ВАЖНО! 
# ADMIN_CHAT_ID должен быть ID администратора (вашего личного чата)
# НЕ устанавливайте сюда ID пользователей бота!
# 
# Чтобы узнать свой ID:
# 1. Напишите боту @userinfobot
# 2. Он покажет ваш ID
# 3. Используйте этот ID как ADMIN_CHAT_ID

# Пример:
# BOT_TOKEN=1234567890:ABCdefGHIjklMNOpqrsTUVwxyz
# OPENAI_API_KEY=sk-1234567890abcdef...
# ADMIN_CHAT_ID=123456789
